Lagrange 插值¶

Lagrange 插值基函数¶

$$ l_k(x)=\frac{(x-x_0)\cdots(x-x_{k-1})(x-x_{k+1})\cdots(x-x_n)}{(x_k-x_0)\cdots(x_k-x_{k-1})(x_k-x_{k+1})\cdots(x_k-x_n)}\quad k=0,1,\cdots,n. $$

$$ l_j(x_k)=\delta_{jk}= \begin{cases} 1,&k=j\\ 0,&k\neq j \end{cases} \quad j,k=0,1,\cdots,n $$

Lagrange 插值多项式¶

$$ L_n(x)=\sum_{k=0}^ny_kl_k(x) $$

三次样条插值¶

$S(x)$$S''(x)$ $$ S''(x)=M_j\frac{x_{j+1}-x}{h_j}+M_{j+1}\frac{x-x_j}{h_j} $$ $S(x_j)=y_j,S(x_{j+1})=y_{j+1}$, $$ S(x)=M_j\frac{(x_{j+1}-x)^3}{6h_j}+M_{j+1}\frac{(x-x_j)^3}{6h_j}+\left(y_j-\frac{M_jh_j^2}6\right)\frac{x_{j+1}-x}{h_j}+\left(y_{j+1}-\frac{M_{j+1}h_j^2}6\right)\frac{x-x_j}{h_j} $$ $S'(x)$

$S'(x_j+0)=S'(x_j-0)$ $$ \mu_jM_{j-1}+2M_j+\lambda_jM_{j+1}=d_j\qquad j=1,2,\cdots,n-1 $$ $$ \mu_j=\frac{h_{j-1}}{h_{j-1}+h_j},\lambda_j=\frac{h_j}{h_{j-1}+h_j}, $$

$$ d_j=6f[x_{j-1},x_j,x_{j+1}]\qquad j=1,2,\cdots,n-1 $$

$$ S''(x_0)=S''(x_n)=0 $$ $$ \lambda_0=\mu_n=0,d_0=d_n=0 $$

$$ \begin{pmatrix} 2&\lambda_0\\ \mu_1&2&\lambda_1\\ &\ddots&\ddots&\ddots\\ &&\mu_{n-1}&2&\lambda_{n-1}\\ &&&\lambda_n&2\\ \end{pmatrix} \begin{pmatrix} M_0\\M_1\\\vdots \\M_{n-1}\\M_n \end{pmatrix} = \begin{pmatrix} d_0\\d_1\\\vdots \\d_{n-1}\\d_n \end{pmatrix} $$

struct PiecewisePolynomial
    piece::Vector{Real}
    polynomials::Vector{Polynomial}
    n::Integer
end

function PiecewisePolynomial(piece, polynomials)
    n = length(piece)
    n = length(polynomials)
    @assert n == n+1 
    return PiecewisePolynomial(piece, polynomials,n) 
end

function (pp::PiecewisePolynomial)(x)
    for i = 1:pp.n
        if x  pp.piece[i+1]
            return pp.polynomials[i](x)
        end
    end
end

function LagrangeBaseFunc(x::Vector,x::Real)::Polynomial
    l = (fromrootsfilter)(x->xx,x)
    l / l(x)
end

function Lagrange(x::Vector,y::Vector)::Polynomial
    bases = @(LagrangeBaseFunc(x,_)).(x)
    return sum(bases.*y)
end;

function Spline(x::Vector,y::Vector;=0,=0,d=0,d=0)
    n = length(x)
    h = x[2:n].-x[1:n-1]
    sh = h[1:n-2].+h[2:n-1]
     = [;h[2:n-1]./sh]
     = [h[1:n-2]./sh;]
    d(i) = 6((y[i+1]-y[i])/h[i]-(y[i]-y[i-1])/h[i-1])
    d = [d;d.(2:n-1)./sh;d]
    M = Tridiagonal(,2ones(n),)\d
    S(j) = (M[j]*Polynomial([x[j+1],-1])^3/6
            + M[j+1]*Polynomial([-x[j],1])^3/6
            + (y[j]-M[j]*h[j]^2/6)*Polynomial([x[j+1],-1])
            + (y[j+1]-M[j+1]*h[j]^2/6)*Polynomial([-x[j],1]))
    S =  S.(1:n-1)./h
    S = PiecewisePolynomial(x,S)
end

function interpolate(method::Function,target::Function,interval::Vector,n::Integer)
    x = range(interval...;length=n+1) |> collect
    y = target.(x)
    return method(x,y)
end;

x = range(-1,1;length=300) |> collect
y = f.(x)

function experiment(;n,legend=true)
    L = interpolate(Lagrange,f,[-1,1],n)
    L = abs.(y .- L.(x))

    S = interpolate(Spline,f,[-1,1],n)
    S = abs.(y .- S.(x))

    plot1 = plot(x,y,label="Runge Function",title="Function Plot(n=$n)")
    plot!(plot1,x,L.(x),label="Lagrange Interpolation")
    plot!(plot1,x,S.(x),label="Cubic Spline Interpolation")
    ylims!(plot1,-.2,1.5)

    plot2 = plot(x,zero(x),label="0",title="Error of Interpolation")
    plot!(plot2,x,L,label="L")
    plot!(plot2,x,S,label="S")
    
    plot(plot1,plot2,layout = (2, 1),legend=legend)  
end;
