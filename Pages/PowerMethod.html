迭代与收敛原理¶

$v_0\neq0$,$x_i$$A$$\lambda_i$$A$ $$ v_0=\sum_{i=1}^n\alpha_i x_i $$ $\{v_k\}$$\{u_k\}$

$$ u_0=v_0 $$$$ v_k=Au_{k-1},u_k=\frac{v_k}{abs \max{\{v_k\}}} $$$$ \begin{aligned} v_k &=\frac{A^kv_0}{abs\max{\{A^{k-1}v_0\}}}\\ &=\frac{\sum_{i=1}^n\alpha_i \lambda_i^k x_i}{abs\max\{\sum_{i=1}^n\alpha_i \lambda_i^{k-1} x_i\}}\\ &=\lambda_1\frac{\alpha_1x_1+\sum_{i=2}^n\alpha_i (\frac{\lambda_i}{\lambda_1})^k x_i}{abs\max\{\alpha_1x_1+\sum_{i=2}^n\alpha_i (\frac{\lambda_i}{\lambda_1})^{k-1} x_i\}}\\ abs\max\{v_k\}&=\frac{\lambda_1abs\max\{\alpha_1x_1+\sum_{i=2}^n\alpha_i (\frac{\lambda_i}{\lambda_1})^k x_i\}}{abs\max\{\alpha_1x_1+\sum_{i=2}^n\alpha_i (\frac{\lambda_i}{\lambda_1})^{k-1} x_i\}}\rightarrow\lambda_1\quad(k\rightarrow \infty) \end{aligned} $$

算法描述¶

$v_0$$\epsilon$ $$ u\leftarrow v_0,\lambda\leftarrow0 $$ loop

  1. $v\leftarrow Au$
  2. $u\leftarrow v/absmax(v)$
  3. $\lambda'\leftarrow absmax(v),\Delta\lambda\leftarrow abs(\lambda'-\lambda)$
  4. if $\Delta\lambda<\epsilon$, then return $\lambda'$
  5. $\lambda\leftarrow\lambda'$

瑞利商加速¶

$$ v_k=Au_{k-1},u_k=\frac{v_k}{abs\max{\{v_k\}}} $$

$\lambda_1$ $$ \lambda_1 \leftarrow \frac{(Au_k,u_k)}{(u_k,u_k)}\quad(k\rightarrow\infty) $$

$\lambda_1$$\lambda_1$

反幂法¶

$p$$(A-pI)^{-1}$ $$ \frac1{\lambda_i-p} $$ $x_i$

$(A-pI)^{-1}$ $$ \begin{cases} u_0=v_0\neq0&v_0\\ v_k=(A-pI)^{-1}u_{k-1}&k=1,2,\cdots\\ u_k=\frac{v_k}{abs\max{\{v_k\}}} \end{cases} $$ $p \rightarrow \lambda_j$ $$ |\lambda_j-p|<<|\lambda_i-p|,\quad i\neq j $$ $$ \left|\frac1{\lambda_j-p}\right|>\left|\frac1{\lambda_i-p}\right|,\quad\forall i\neq j $$ $\frac1{\lambda_j-p}$ $$ \frac1{\lambda_j-p}\leftarrow abs\max\{v_k\},\quad(k\rightarrow\infty) $$ $$ \lambda_j\leftarrow p+\frac1{abs\max\{v_k\}},\quad(k\rightarrow\infty) $$

absmax(v) = maximum(abs,v)
norm!(v) = v ./ absmax(v)

PowerExpr(A,u,v) = absmax(v)
RayleighExpr(A,u,v) = sum(A*u.*u)/sum(u.*u)

function MainEigen(A::Matrix;expr=PowerExpr,u=NaN,digits=4,maxiter=100,reqs=false)
    n = size(A)[1]
    u == NaN || (u = ones(n))  
     = 0.
    reqs && (s = Vector{Float64}())
    for i  1:maxiter
        v = A*u
        u = norm!(v)
         = expr(A,u,v)
        ok = false
        abs(-) < .1^digits && (ok = true)
         = 
        reqs && push!(s,)
        if ok
            if reqs
                return ,u,s
            end
            return ,u
        end
    end
    (throwerror)("Can't find solution satisfied precise demand within the maximum number of iterations.")
end

function InvPowMethod(A::Matrix,p::Number;expr=PowerExpr,u=NaN,digits=4,maxiter=100,reqs=false)
    n = size(A)[1]
    u == NaN || (u = ones(n))
    B = inv(A - p*I)
     = 0.
    reqs && (s = Vector{Float64}())
    for i  1:maxiter
        v = B*u
        u = norm!(v)
         = p+1/expr(B,u,v)
        ok = false
        abs(-) < .1^digits && (ok = true)
         = 
        reqs && push!(s,)
        if ok
            if reqs
                return ,u,s
            end
            return ,u
        end
    end
    (throwerror)("Can't find solution satisfied precise demand within the maximum number of iterations.")
end
